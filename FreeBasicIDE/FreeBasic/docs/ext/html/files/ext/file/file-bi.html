Examples

Simple File class example

#include "ext/file.bi"
using ext

var myfile = ext.File("test.txt",R)

if myfile.open = false then

  do while not myfile.eof

      print myfile.readLine

  loop

else

  print GetErrorText(myfile.LastError)

end if
